318 void keyboard(unsigned char c, int x, int y)
319 {
320  switch (c)
321  {
322  case 'w': //GO STRAIGHT
323  if(cam == 0){
324  cameraVectorGeneral.x += sin(angle);
325  cameraVectorGeneral.z += cos(angle);
326  }else{
327  position_rocket = VectorAdd(position_rocket, forwardVector);
328  }
329  break;
330  case 's': //GO BACKWARDS
331  if(cam == 0){
332  cameraVectorGeneral.x -= sin(angle);
333  cameraVectorGeneral.z -= cos(angle);
334  }else{
335  position_rocket = VectorSub(position_rocket, forwardVector);
336  }
337  break;
338  case 'a': //TURN LEFT
339  if(cam == 0){
340  angle = fmod(angle + 0.03 + 2*M_PI, 2*M_PI);
341  }else{
342  forwardVector = Normalize(VectorAdd(forwardVector, ScalarMult(rightVector, -ROTATION_DEGREE)));
343  rightVector = Normalize(CrossProduct(forwardVector, upVector));
344  }
345  break;
346  case 'd': //TURN RIGHT
347  if(cam == 0){
348  angle = fmod(angle - 0.03 + 2*M_PI, 2*M_PI);
349  }else{
350  forwardVector = Normalize(VectorAdd(forwardVector, ScalarMult(rightVector, ROTATION_DEGREE)));
351  rightVector = Normalize(CrossProduct(forwardVector, upVector));
352  }
353  break;
354  case '+': //ROTATE UP
355  if(cam == 1){
356  forwardVector = Normalize(VectorAdd(forwardVector, ScalarMult(upVector, ROTATION_DEGREE)));
357  upVector = Normalize(CrossProduct(rightVector, forwardVector));
358  }
359  break;
360  case '-': //ROTATE DOWN
361  if(cam == 1){
362  forwardVector = Normalize(VectorAdd(forwardVector, ScalarMult(upVector, -ROTATION_DEGREE)));
363  upVector = Normalize(CrossProduct(rightVector, forwardVector));
364  }
365  break;
366  case 'q': //CHANGE CAMERA VIEW
367  cam = fmod(cam + 1, 2);
368  break;
369  }
370 
371  if(cam == 0){
372  viewVectorGeneral.x = cameraVectorGeneral.x + sin(angle);
373  viewVectorGeneral.z = cameraVectorGeneral.z + cos(angle);
374  }else{
375  cameraDirectionVector = Normalize(VectorAdd(ScalarMult(forwardVector, -1), ScalarMult(upVector, 0.5)));
376  position_camera = VectorAdd(position_rocket, ScalarMult(cameraDirectionVector, 3.0));
377  rotation_matrix = Transpose(SetMatrix4(rightVector.x, rightVector.y, rightVector.z, 0,
378  upVector.x, upVector.y, upVector.z, 0,
379  forwardVector.x, forwardVector.y, forwardVector.z, 0,
380  0, 0, 0, 1));
381  }
382  glutPostRedisplay();
383 }
